LibraryImport.Cobol

// Test LinkagePool declaration
LinkagePool.TEST_LINKAGE {
    "field1": Initialize=10
    "field2": Initialize="hello"
    "field3": Initialize=0
}

SubRoutine.Main {
    // Test 1: Direct assignment from LinkagePool field
    val1 = TEST_LINKAGE.field1
    PrintMessage("Direct access field1: ")
    PrintNumber(val1)
    PrintMessage("\n")
    
    // Test 2: Direct use in condition
    IfCondition EqualTo(TEST_LINKAGE.field1, 10) ThenBlock: {
        PrintMessage("Direct condition works!\n")
    } ElseBlock: {
        PrintMessage("Direct condition FAILED\n")
    }
    
    // Test 3: Assignment to LinkagePool field
    TEST_LINKAGE.field3 = 42
    PrintMessage("After assignment, field3: ")
    PrintNumber(TEST_LINKAGE.field3)
    PrintMessage("\n")
    
    // Test 4: String field access
    str_val = TEST_LINKAGE.field2
    PrintMessage("String field: ")
    PrintString(str_val)
    PrintMessage("\n")
    
    ReturnValue(0)
}

RunTask(Main)
